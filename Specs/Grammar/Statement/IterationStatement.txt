IterationStatement
	: WhileStatement
	| DoWhileStatement
	| ForStatement
	;

WhileStatement
	: 'while' '(' Expression ')' Statement
	;

DoWhileStatement
	: 'do' Statement 'while' '(' Expression ')' ';'
	;

ForStatement
	: 'for' '(' OptionalForVariableInitializer ';' OptionalExpression ';' OptionalExpression ')' Statement
	;

ForVariableInitializer
	: VariableStatement
	| Expression
	;
